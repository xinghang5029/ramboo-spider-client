ID INTEGER PRIMARY KEY AUTOINCREMENT,
NAME CHAR(256),
URL CHAR(1024),
THEME CHAR(10),
CATEGORY CHAR(10),

USER CHAR(20),
STATE INT(1)
)

CREATE TABLE rb_theme(
ID INT PRIMARY KEY NOT NULL,
NAME CHAR(256),
CODE CHAR(256)

)

CREATE TABLE rb_category(
ID INT PRIMARY KEY NOT NULL,
NAME CHAR(256),
CODE CHAR(256),
THEME CHAR(256),
PARENT CHAR(256)

)
